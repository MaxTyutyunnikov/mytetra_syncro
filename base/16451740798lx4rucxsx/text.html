Получить адрес функции в виде числа в языке Си++ можно через приведение указателя на функцию к нужному числовому типу. Делается это следующим образом:



// Код функции boo() находится в ячейке, например, памяти 002B1050

int boo()

{

return 7;

}

int main()

{

    unsigned long a = reinterpret_cast<unsigned long>( boo );
   printf("a = %X\n", a);

return 0;

}


В консоль будет выведено:


002B1050

002B1050


Доработать


#include <iostream>
#include <stdio.h>

// Код функции boo() находится в ячейке, например, памяти 002B1050
int boo() 
{
   return 7;
}

int main()
{
 
   unsigned long a = reinterpret_cast<unsigned long>( boo );
   printf("a = %X\n", a);

   std::cout << boo;
   std::cout << std::endl;

   printf("%X\n", boo);

   std::cout << reinterpret_cast<void*>(boo); 

   return 0;
}
