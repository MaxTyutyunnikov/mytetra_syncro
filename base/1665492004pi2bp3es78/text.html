Нужно настроить бит SWJ_CFG регистра AFIO_MAPR для отключения JTAG, оставьте только порт отладки SWD для отладки вашего кода


// Закрываем JTAG для выпуска PB3, PB4, PA15

AFIO->MAPR |= AFIO_MAPR_SWJ_CFG_JTAGDISABLE;


// Дополнительная настройка для освобождения PA15 - RCC_APB2ENR_AFIOEN

RCC->APB2ENR = RCC_APB2ENR_USART1EN | RCC_APB2ENR_TIM1EN | RCC_APB2ENR_SPI1EN \

| RCC_APB2ENR_ADC1EN | RCC_APB2ENR_ADC2EN | RCC_APB2ENR_AFIOEN \

| RCC_APB2ENR_IOPDEN | RCC_APB2ENR_IOPCEN | RCC_APB2ENR_IOPBEN | RCC_APB2ENR_IOPAEN;
