Ниже дан пример скрипта, который перебирает *.jpg-файлы в директории и переименовывает их сквозной нумерацией. Скрипт корректно обрабатывает файлы с именами, в которых присутсвуют пробелы:



#!/bin/bash


n=1


ls -1 *.jpg | while read file

do


echo "Обрабатывается файл: $file"

numName=$(printf %03d $n)


mv -f "$file" $numName.jpg

n=$(($n+1))

done
