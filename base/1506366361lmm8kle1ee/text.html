У прошивки DD-WRT есть следующая проблема: если войти в веб-интерфейс, и отключить


TP-Link TL-WR1043ND



root@DD-WRT:~# nvram set httpd_enable=1

root@DD-WRT:~# nvram set http_enable=1

root@DD-WRT:~# nvram set http_lanport=80

root@DD-WRT:~# nvram set httpsd_enable=0

root@DD-WRT:~# nvram set https_enable=0

root@DD-WRT:~# nvram set http_wanport=8080

root@DD-WRT:~# nvram commit

root@DD-WRT:~# reboot
