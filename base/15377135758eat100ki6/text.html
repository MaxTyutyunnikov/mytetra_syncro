Чтобы узнать системную кодировку в Windows, надо понимать, что в Qt нет готовых методов. Поэтому придется делать все с помощью функции WinAPI и условной компиляцией.


Код будет выглядеть примерно так:



#ifdef Q_OS_WIN32

#include "windows.h"

#endif


using namespace std;


QString getSystemCodepage()

{

#ifdef Q_OS_WIN32

return "CP "+QString::number( GetACP() );

#endif


#ifdef Q_OS_LINUX

return "UTF-8");

#endif


return "";

}



А использовать эту функцию можно примерно так:


QString codePage=getSystemCodepage();

qDebug() << "System code page: " << codePage;

if(codePage.size()>0) {

QTextCodec::setCodecForLocale(QTextCodec::codecForName(codePage.toLatin1()));

}
