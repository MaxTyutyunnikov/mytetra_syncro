Давайте обратимся к нескольким примерам.  Первый - простая инициализация для приложения с одним кадром: 

#include <GL/glut.h> 

void main(int argcp, char **argv){ 

/* Установка размеров и положения окна */ 
glutInitWindowSize(640, 480); 
glutInitWindowPosition(0, 0); 

/* Выбор режима окна:  
   Одиночный буфер и RGBA палитра */ 
glutInitDisplayMode(GLUT_RGBA | GLUT_SINGLE); 

/* Инициализация первоначального состояния GLUT */ 
glutInit(&argcp, argv);  

.....немного кода; 

};
